            IF @TtRcvTcQtuToReceive > @ItemlocQtyOnHand - @ItemlocQtyRsvd - @ItemlocQtyContained   
            and (@ItemlocQtyRsvd > 0 or @ItemlocQtyContained > 0)
            BEGIN /* DEPTH:5 */
                EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=NoCompare'
                   , '@itemlifo.qty'
                   , @TtRcvTcQtuToReceive

                EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=IsCompare3'

                   , @ItemlocItem

                IF @ItemlocQtyRsvd > 0

                    , '@itemloc.item'
                    , @ItemlocItem

                    , @ItemlocLoc

                  EXEC dbo.MsgAskSp @Infobar OUTPUT, @PromptButtons OUTPUT

                    , '@!GoToContainer'

            END /* DEPTH:5 */

            IF (@ContainsOnlyTaxFreeMatls = 1 AND @TaxFreeMatl = 1) OR (@TaxedMatl = 1 AND @MexicanCountryPackOn = 1)
            BEGIN /* DEPTH:5 */
               if @ItemLotTracked = 1
                  SELECT
                    @TFIQRowPointer = tfiq.RowPointer
                  , @TFIQQtyOnHand  = tfiq.qty_on_hand

                  SELECT
                    @TFIQRowPointer = tfiq.RowPointer
                  , @TFIQQtyOnHand  = tfiq.qty_on_hand
                  , @TFIQQtyRsvd    = tfiq.qty_rsvd
                  FROM  tax_free_import_qty AS tfiq with (readuncommitted)

                  AND   tfiq.item = @PoitemItem
                  AND   tfiq.lot is null

                  AND   tfiq.import_doc_id  = @TtRcvImportDocId

               if @@rowcount <> 1
                  set @TFIQRowPointer = null

               IF (@TFIQRowPointer is not null)
               BEGIN /* DEPTH:6 */
                  if (@TFIQQtyOnHand - @TFIQQtyRsvd ) < @TtRcvTcQtuToReceive AND @TFIQQtyRsvd > 0
                    BEGIN /* DEPTH:7 */
                     EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=NoCompare'

               end /* DEPTH:6 */
            end /* DEPTH:5 */ /* Tax Free Imported Material */

         end /* DEPTH:4 */  /* available item and available itemloc */
      end /* DEPTH:3 */  /* if tt-rcv.tc-qtu-to-receive > 0.0 then do: */

      else

         if @QtyReturned + @TtRcvTcQtuToReceive < 0.0

               , @TMText
               , '@tmp_rcv'
               , '@tmp_rcv.po_num'
               , @TtRcvPoNum
               , '@tmp_rcv.po_line'

               , '@tmp_rcv.po_release'
               , @TtRcvPoRelease
               , '@grn_line.grn_num'

         if @GrnLineLoop > 0
         BEGIN /* DEPTH:4 */

             @LotLocRowpointer = lot_loc.rowpointer

            and @LotLocQtyOnHand - @TtRcvTcQtuToReceive < 0.0
            BEGIN /* DEPTH:5 */
               IF @InvparmsNegFlag = 0
               BEGIN /* DEPTH:6 */
                  /* Negative quantities not allowed */
                  EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=CmdFailed3'

                     , '@tmp_rcv'

                     , @TtRcvTcQtuToReceive

                     , @PoitemItem
                     , '@lot_loc.loc'
                     , @TtRcvLoc
                     , '@lot_loc.lot'
                     , @TtRcvLot

                  , '@lot_loc.qty_on_hand', @MsgParm1

         end /* DEPTH:4 */ /* item lot tracked */


         IF (@ContainsOnlyTaxFreeMatls = 1 AND @TaxFreeMatl = 1) OR (@TaxedMatl = 1 AND @MexicanCountryPackOn = 1)
            BEGIN /* DEPTH:4 */
               if @ItemLotTracked = 1
                  SELECT
                    @TFIQRowPointer = tfiq.RowPointer
                  , @TFIQQtyOnHand  = tfiq.qty_on_hand
                  , @TFIQQtyRsvd    = tfiq.qty_rsvd
                  FROM  tax_free_import_qty AS tfiq with (readuncommitted)
                  WHERE tfiq.whse = @Currentwhse
                  AND   tfiq.item = @PoitemItem
                  AND   tfiq.lot  = @TtRcvLot
                  AND   tfiq.loc  = @TtRcvLoc
                  AND   tfiq.import_doc_id  = @TtRcvImportDocId
               else
                  SELECT
                    @TFIQRowPointer = tfiq.RowPointer
                  , @TFIQQtyOnHand  = tfiq.qty_on_hand
                  , @TFIQQtyRsvd    = tfiq.qty_rsvd
                  FROM  tax_free_import_qty AS tfiq with (readuncommitted)
                  WHERE tfiq.whse = @Currentwhse
                  AND   tfiq.item = @PoitemItem
                  AND   tfiq.lot is null
                  AND   tfiq.loc  = @TtRcvLoc
                  AND   tfiq.import_doc_id  = @TtRcvImportDocId

               if @@rowcount <> 1
                  set @TFIQRowPointer = null

               IF (@TFIQRowPointer is not null) and
               (@TFIQQtyOnHand -@TtRcvTcQtuToReceive) < 0
                    BEGIN /* DEPTH:5 */
                  /* Negative quantities not allowed */
                    EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=CmdFailed3'
                     , @TMText
                     , '@tmp_rcv'
                     , '@tmp_rcv.po_num'
                     , @TtRcvPoNum
                     , '@tmp_rcv.po_line'
                     , @TtRcvPoLine
                     , '@tmp_rcv.po_release'
                     , @TtRcvPoRelease

                    EXEC dbo.MsgAppSp @Infobar OUTPUT, 'I=IsCompare='
                     , '@tmp_rcv.qty_to_receive'
                     , @TtRcvTcQtuToReceive

                     EXEC dbo.MsgAppSp @Infobar OUTPUT, 'I=IsCompare5'

                        , '@tax_free_import_qty.whse'

                        , @TtRcvImportDocId

                     EXEC @MsgSeverity = dbo.MsgAppSp @Infobar OUTPUT, 'E=NoCompare='
                     , '@tax_free_import_qty.qty_on_hand', @MsgParm1
                     IF @MsgSeverity >= ISNULL(@Severity, 0)
                        SET @Severity = @MsgSeverity

                     GOTO LineError
                  end /* DEPTH:5 */
            end /* DEPTH:4 */ /* Tax Free Imported Material */


      end /* DEPTH:3 */ /* itemloc available */
   end /* DEPTH:2 */ /* if tt-rcv.dr-return */

   /* Receiving */
   else
   BEGIN /* DEPTH:2 */
      if @TtRcvTcQtuToReceive > 0.0
      BEGIN /* DEPTH:3 */
         if (@ItemRowpointer is not null) AND @ItemSerialTracked <> 0
         BEGIN /* DEPTH:4 */
            set @TmpSerSerNum = null
            select top 1
              @TmpSerSerNum = tmp_ser.ser_num
            , @SerialStat = serial.stat
            from tmp_ser with (readuncommitted)
               inner join serial with (readuncommitted) on
                  serial.ser_num = tmp_ser.ser_num

            where tmp_ser.SessionID = @SessionID
            and tmp_ser.ref_str = @TtRcvWorkkey

            begin
               SET @MsgParm1 = '@:SerialStatus:' + @SerialStat

               IF @MsgSeverity >= ISNULL(@Severity, 0)
                  SET @Severity = @MsgSeverity

               GOTO LineError
            end
         end /* DEPTH:4 */  /* if avail item and item.serial-tracked */
      end /* DEPTH:3 */

      /* receiving negative quantity */
      else
      BEGIN /* DEPTH:3 */
         if @QtyReceived - (@QtyVoucher - @QtyReturned) + @TtRcvTcQtuToReceive < 0
         BEGIN /* DEPTH:4 */
            EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=CmdFailed4'

               , '@tmp_rcv'
               , '@tmp_rcv.po_num'
               , @TtRcvPoNum
               , '@tmp_rcv.po_line'
               , @TtRcvPoLine
               , '@tmp_rcv.po_release'
               , @TtRcvPoRelease
               , '@grn_line.grn_num'
               , @GrnLineGrnNum

            EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=IsCompare'
               , '@poitem.qty_received'
               , @QtyReceived

            SET @MsgParm1 = @QtyVoucher - @QtyReturned

            EXEC @MsgSeverity = dbo.MsgAppSp @Infobar OUTPUT, 'E=IsCompare'
            , '@poitem.qty_voucher', @MsgParm1
            IF @MsgSeverity >= ISNULL(@Severity, 0)
               SET @Severity = @MsgSeverity

            SET @MsgParm1 = - (@QtyReceived - (@QtyVoucher - @QtyReturned))

            EXEC @MsgSeverity = dbo.MsgAppSp @Infobar OUTPUT, 'E=NoCompare<'
            , '@tmp_rcv.qty_to_receive_conv', @MsgParm1
            IF @MsgSeverity >= ISNULL(@Severity, 0)
               SET @Severity = @MsgSeverity

            set @NeedGrnInError = 1
            GOTO LineError
         end /* DEPTH:4 */ /* voucher exceeded check */
         if @GrnLineLoop > 0
         BEGIN /* DEPTH:4 */
            if @GrnLineQtyRec - (@GrnLineQtyVouchered - @GrnLineQtyReturned) + @TtRcvTcQtuToReceive < 0
            BEGIN /* DEPTH:5 */
               EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=CmdFailed3'
                  , @TMText
                  , '@tmp_rcv'
                  , '@grn_line.vend_num'
                  , @GrnLineVendNum
                  , '@grn_line.grn_num'
                  , @GrnLineGrnNum
                  , '@grn_line.grn_line'
                  , @GrnLineGrnLine

               EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=IsCompare'
                  , '@grn_line.qty_rec'
                  , @GrnLineQtyRec

               SET @MsgParm1 = @GrnLineQtyVouchered - @GrnLineQtyReturned

               EXEC @MsgSeverity = dbo.MsgAppSp @Infobar OUTPUT, 'E=IsCompare'
               , '@grn_line.qty_vouchered', @MsgParm1
               IF @MsgSeverity >= ISNULL(@Severity, 0)
                  SET @Severity = @MsgSeverity

               SET @MsgParm1 = - (@GrnLineQtyRec - (@GrnLineQtyVouchered - @GrnLineQtyReturned))

               EXEC @MsgSeverity = dbo.MsgAppSp @Infobar OUTPUT, 'E=NoCompare<'
               , '@tmp_rcv.qty_to_receive_conv', @MsgParm1
               IF @MsgSeverity >= ISNULL(@Severity, 0)
                  SET @Severity = @MsgSeverity

               GOTO LineError
            end /* DEPTH:5 */  /* voucher exceeded check */
         end /* DEPTH:4 */  /* if tt-rcv.grn-line > 0 */

         if (@ItemlocRowpointer is not null)
         BEGIN /* DEPTH:4 */
            if @ItemlocQtyOnHand + @TtRcvTcQtuToReceive < 0.0
            BEGIN /* DEPTH:5 */
               IF @InvparmsNegFlag = 0
               BEGIN /* DEPTH:6 */
                  /* ALLOW NEGATIVE QTY */
                  EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=CmdFailed3'
                     , @TMText
                     , '@tmp_rcv'
                     , '@tmp_rcv.po_num'
                     , @TtRcvPoNum
                     , '@tmp_rcv.po_line'
                     , @TtRcvPoLine
                     , '@tmp_rcv.po_release'
                     , @TtRcvPoRelease

                  EXEC dbo.MsgAppSp @Infobar OUTPUT, 'I=IsCompare='
                     , '@tmp_rcv.qty_to_receive'
                     , @TtRcvTcQtuToReceive

                  EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'I=IsCompare=3'
                     , '@itemloc.qty_on_hand'
                     , @ItemlocQtyOnHand
                     , '@itemloc'
                     , '@itemloc.whse'
                     , @Currentwhse
                     , '@itemloc.item'
                     , @PoitemItem
                     , '@itemloc.loc'
                     , @TtRcvLoc

                  SET @MsgParm1 = @ItemlocQtyOnHand + @TtRcvTcQtuToReceive

                  EXEC @MsgSeverity = dbo.MsgAppSp @Infobar OUTPUT, 'E=NoCompare='
                  , '@itemloc.qty_on_hand', @MsgParm1
                  IF @MsgSeverity >= ISNULL(@Severity, 0)
                     SET @Severity = @MsgSeverity

                  GOTO LineError
               end /* DEPTH:6 */
            end /* DEPTH:5 */  /* if itemloc.qty-on-hand + tt-rcv.tc-qtu-to-receive < 0.0 */

            IF @ItemLotTracked <> 0
            BEGIN /* DEPTH:5 */
               select
                @LotLocRowpointer = lot_loc.rowpointer
               ,@LotLocQtyOnHand = lot_loc.qty_on_hand
               ,@LotLocQtyRsvd = lot_loc.qty_rsvd
               ,@LotLocQtyContained = lot_loc.qty_contained
               from lot_loc
               where lot_loc.whse = @Currentwhse
                  and lot_loc.item = @PoitemItem
                  and lot_loc.loc = @TtRcvLoc
                  and lot_loc.lot = @TtRcvLot

               if @@rowcount <> 1
                  set @LotLocRowpointer = null

               if (@LotLocRowpointer is not null)
               BEGIN /* DEPTH:6 */
                   if ABS(@TtRcvTcQtuToReceive) > @LotLocQtyOnHand - @LotLocQtyRsvd - @LotLocQtyContained
                   and (@LotLocQtyRsvd > 0 or @LotLocQtyContained > 0)
                    BEGIN /* DEPTH:7 */
                     EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=NoCompare'
                        , '@itemlifo.qty'
                        , @TtRcvTcQtuToReceive

                     EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=IsCompare4'
                        , '@lot_loc.qty_on_hand'
                        , @LotLocQtyOnHand
                        , '@lot_loc'
                        , '@lot_loc.whse'
                        , @Currentwhse
                        , '@lot_loc.item'
                        , @PoitemItem
                        , '@lot_loc.loc'
                        , @TtRcvLoc
                        , '@lot_loc.lot'
                        , @TtRcvLot
                        
                     IF @LotLocQtyRsvd > 0
                        EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=IsCompare4'
                            , '@lot_loc.qty_rsvd'
                            , @LotLocQtyRsvd
                            , '@lot_loc'
                            , '@lot_loc.whse'
                            , @Currentwhse
                            , '@lot_loc.item'
                            , @PoitemItem
                            , '@lot_loc.loc'
                            , @TtRcvLoc
                            , '@lot_loc.lot'
                            , @TtRcvLot
                            
                     IF @LotLocQtyContained > 0
                     BEGIN
                        EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=IsCompare4'
                            , '@lot_loc.qty_contained'
                            , @LotLocQtyContained
                            , '@lot_loc'
                            , '@lot_loc.whse'
                            , @Currentwhse
                            , '@lot_loc.item'
                            , @PoitemItem
                            , '@lot_loc.loc'
                            , @TtRcvLoc
                            , '@lot_loc.lot'
                            , @TtRcvLot
                            
                        EXEC dbo.MsgAskSp @Infobar OUTPUT, @PromptButtons OUTPUT
                          , 'Q=FunctPerformNoYes'
                          , '@!GoToContainer'
                     END
                            
                     GOTO LineError
                  end /* DEPTH:7 */

                  if @LotLocQtyOnHand + @TtRcvTcQtuToReceive < 0.0
                  BEGIN /* DEPTH:7 */
                     IF @InvparmsNegFlag = 0
                     BEGIN /* DEPTH:8 */
                        /* Negative quantities not allowed */
                        EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=CmdFailed3'
                           , @TMText
                           , '@tmp_rcv'
                           , '@tmp_rcv.po_num'
                           , @TtRcvPoNum
                           , '@tmp_rcv.po_line'
                           , @TtRcvPoLine
                           , '@tmp_rcv.po_release'
                           , @TtRcvPoRelease

                        EXEC dbo.MsgAppSp @Infobar OUTPUT, 'I=IsCompare='
                           , '@tmp_rcv.qty_to_receive'
                           , @TtRcvTcQtuToReceive

                        EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'I=IsCompare=4'
                           , '@lot_loc.qty_on_hand'
                           , @LotLocQtyOnHand
                           , '@lot_loc'
                           , '@lot_loc.whse'
                           , @Currentwhse
                           , '@lot_loc.item'
                           , @PoitemItem
                           , '@lot_loc.loc'
                           , @TtRcvLoc
                           , '@lot_loc.lot'
                           , @TtRcvLot

                        SET @MsgParm1 = @LotLocQtyOnHand + @TtRcvTcQtuToReceive

                        EXEC @MsgSeverity = dbo.MsgAppSp @Infobar OUTPUT, 'E=NoCompare='
                        , '@lot_loc.qty_on_hand', @MsgParm1
                        IF @MsgSeverity >= ISNULL(@Severity, 0)
                           SET @Severity = @MsgSeverity

                        GOTO LineError
                     end /* DEPTH:8 */ /* negative quantities not allowed */
                  end /* DEPTH:7 */ /* lot-loc quantity being driven negative */
               end /* DEPTH:6 */ /* lot-loc available */
            end /* DEPTH:5 */ /* item lot tracked */
            
            if ABS(@TtRcvTcQtuToReceive) > @ItemlocQtyOnHand - @ItemlocQtyRsvd - @ItemlocQtyContained
            and (@ItemlocQtyRsvd > 0 or @ItemlocQtyContained > 0)
            BEGIN /* DEPTH:5 */
               EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=NoCompare'
                  , '@itemlifo.qty'
                  , @TtRcvTcQtuToReceive

               EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=IsCompare3'
                  , '@itemloc.qty_on_hand'
                  , @ItemlocQtyOnHand
                  , '@itemloc'
                  , '@itemloc.whse'
                  , @ItemlocWhse
                  , '@itemloc.item'
                  , @ItemlocItem
                  , '@itemloc.loc'
                  , @ItemlocLoc

               IF @ItemlocQtyRsvd > 0
                  EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=IsCompare3'
                      , '@itemloc.qty_rsvd'
                      , @ItemlocQtyRsvd
                      , '@itemloc'
                      , '@itemloc.whse'
                      , @ItemlocWhse
                      , '@itemloc.item'
                      , @ItemlocItem
                      , '@itemloc.loc'
                      , @ItemlocLoc

               IF @ItemlocQtyContained > 0
               BEGIN
                  EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=IsCompare3'
                      , '@itemloc.qty_contained'
                      , @ItemlocQtyContained
                      , '@itemloc'
                      , '@itemloc.whse'
                      , @ItemlocWhse
                      , '@itemloc.item'
                      , @ItemlocItem
                      , '@itemloc.loc'
                      , @ItemlocLoc
                      
                  EXEC dbo.MsgAskSp @Infobar OUTPUT, @PromptButtons OUTPUT
                    , 'Q=FunctPerformNoYes'
                    , '@!GoToContainer'
               END
               
               GOTO LineError
            end /* DEPTH:5 */


            IF (@ContainsOnlyTaxFreeMatls = 1 AND @TaxFreeMatl = 1) OR (@TaxedMatl = 1 AND @MexicanCountryPackOn = 1)
            BEGIN /* DEPTH:5 */
               if @ItemLotTracked = 1
                  SELECT
                    @TFIQRowPointer = tfiq.RowPointer
                  , @TFIQQtyOnHand  = tfiq.qty_on_hand
                  , @TFIQQtyRsvd    = tfiq.qty_rsvd
                  FROM  tax_free_import_qty AS tfiq with (readuncommitted)
                  WHERE tfiq.whse = @Currentwhse
                  AND   tfiq.item = @PoitemItem
                  AND   tfiq.lot  = @TtRcvLot
                  AND   tfiq.loc  = @TtRcvLoc
                  AND   tfiq.import_doc_id  = @TtRcvImportDocId
               else
                  SELECT
                    @TFIQRowPointer = tfiq.RowPointer
                  , @TFIQQtyOnHand  = tfiq.qty_on_hand
                  , @TFIQQtyRsvd    = tfiq.qty_rsvd
                  FROM  tax_free_import_qty AS tfiq with (readuncommitted)
                  WHERE tfiq.whse = @Currentwhse
                  AND   tfiq.item = @PoitemItem
                  AND   tfiq.lot is null
                  AND   tfiq.loc  = @TtRcvLoc
                  AND   tfiq.import_doc_id  = @TtRcvImportDocId

               if @@rowcount <> 1
                  set @TFIQRowPointer = null

               IF (@TFIQRowPointer is not null)
               BEGIN /* DEPTH:6 */
                  if (@TFIQQtyOnHand - @TFIQQtyRsvd ) < ABS(@TtRcvTcQtuToReceive) AND @TFIQQtyRsvd > 0
                    BEGIN /* DEPTH:7 */
                     EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=NoCompare'
                        , '@itemlifo.qty'
                        , @TtRcvTcQtuToReceive

                     EXEC dbo.MsgAppSp @Infobar OUTPUT, 'I=IsCompare5'
                        , '@tax_free_import_qty.qty_on_hand'
                        , @TFIQQtyOnHand
                        , '@lot_loc'
                        , '@tax_free_import_qty.whse'
                        , @Currentwhse
                        , '@tax_free_import_item.item'
                        , @PoitemItem
                        , '@tax_free_import_qty.loc'
                        , @TtRcvLoc
                        , '@tax_free_import_qty.lot'
                        , @TtRcvLot
                        , '@rsvd_inv.import_doc_id'
                        , @TtRcvImportDocId

                     EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=IsCompare0'
                        , '@lot_loc.qty_rsvd'
                        , @TFIQQtyRsvd
                        , '@lot_loc'

                     GOTO LineError
                  END /* DEPTH:7 */

                  IF (@TFIQQtyOnHand + @TtRcvTcQtuToReceive) < 0
                     BEGIN /* DEPTH:7 */
                    IF @InvparmsNegFlag = 0
                     BEGIN /* DEPTH:8 */
                        /* Negative quantities not allowed */
                      EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=CmdFailed3'
                        , @TMText
                        , '@tmp_rcv'
                        , '@tmp_rcv.po_num'
                        , @TtRcvPoNum
                        , '@tmp_rcv.po_line'
                        , @TtRcvPoLine
                        , '@tmp_rcv.po_release'
                        , @TtRcvPoRelease

                       EXEC dbo.MsgAppSp @Infobar OUTPUT, 'I=IsCompare='
                        , '@tmp_rcv.qty_to_receive'
                        , @TtRcvTcQtuToReceive

                       EXEC dbo.MsgAppSp @Infobar OUTPUT, 'I=IsCompare5'
                        , '@tax_free_import_qty.qty_on_hand'
                        , @TFIQQtyOnHand
                        , '@lot_loc'
                        , '@tax_free_import_qty.whse'
                        , @Currentwhse
                        , '@tax_free_import_item.item'
                        , @PoitemItem
                        , '@tax_free_import_qty.loc'
                        , @TtRcvLoc
                        , '@tax_free_import_qty.lot'
                        , @TtRcvLot
                        , '@rsvd_inv.import_doc_id'
                        , @TtRcvImportDocId

                       SET @MsgParm1 = @TFIQQtyOnHand - @TtRcvTcQtuToReceive

                       EXEC @MsgSeverity = dbo.MsgAppSp @Infobar OUTPUT, 'E=NoCompare='
                        , '@tax_free_import_qty.qty_on_hand', @MsgParm1
                       IF @MsgSeverity >= ISNULL(@Severity, 0)
                          SET @Severity = @MsgSeverity

                       GOTO LineError
                     END /* DEPTH:8 */ /* negative quantities not allowed */
                   END /* DEPTH:7 */
               END /* DEPTH:6 */
            END /* DEPTH:5 */ /* Tax Free Imported Material */


         end /* DEPTH:4 */ /* itemloc available */
      end /* DEPTH:3 */  /* else if tt-rcv.tc-qtu-to-receive < 0.0 then do: */
   end /* DEPTH:2 */

   /* save serials for xref transactions (see POReceivingProjShipSp) */
   DECLARE @PoitemRefNum CoJobProjTrnNumType 
   , @SourceSite SiteType
   , @SourceCoNum CoNumType

   SELECT @PoitemRefNum = poitem.ref_num
   , @SourceCoNum = po.source_site_co_num
   , @SourceSite = vendor.source_site
   FROM poitem
   LEFT OUTER JOIN po ON
      po.po_num = poitem.po_num
   LEFT OUTER JOIN vendor ON
      vendor.vend_num = po.vend_num
   WHERE poitem.po_num = @TtRcvPoNum
     AND poitem.po_line = @TtRcvPoLine
     AND poitem.po_release = @TtRcvPoRelease

   --If the PO Line doesn't have Xref, then don't insert into tmp_ser.
   IF @PoitemRefNum IS NOT NULL AND @PoitemRefNum <> ''
   BEGIN
      INSERT INTO tmp_ser (SessionID, ser_num, ref_str)
      SELECT SessionID, ser_num, 'xref' + @TtRcvWorkkey
      FROM tmp_ser with (readuncommitted)
      WHERE tmp_ser.SessionID = @SessionID
      AND tmp_ser.ref_str = @TtRcvWorkkey
   END

   /* POST TRANSACTION */

   DECLARE
      @HaveItem bit
      , @Cost     CostPrcType
      , @TranCost CostPrcType

   IF @Poinclude_tax_in_cost = 1
   BEGIN
      if @Today is null
         set @Today = dbo.GetSiteDate(getdate())

      if @IsTaxInterfaceAvailable is null
         set @IsTaxInterfaceAvailable = dbo.IsAddonAvailable('TaxInterface')

      if @vrtx_parm_Exists is null
         set @vrtx_parm_Exists = case when OBJECT_ID(N'vrtx_parm') IS NOT NULL then 1 else 0 end

      if @IsJapanInterfaceAvailable is null
         set @IsJapanInterfaceAvailable = dbo.IsAddonAvailable('JapanCountryPack')

      if @DomCurrCode is null
         select @DomCurrCode = curr_code from currparms with (readuncommitted)

      IF @CurrCode IS NULL
         set @CurrCode = @DomCurrCode

      SELECT @CurrencyPlaces = currency.places_cp
      FROM currency with (readuncommitted)
      WHERE  currency.curr_code = @CurrCode

     EXEC @Severity = dbo.TaxPriceSeparationSp
           @InvType                = 'PO'
         , @Type                   = 'I'
         , @TaxCode1               = @PoitemTaxCode1
         , @TaxCode2               = @PoitemTaxCode2
         , @HdrTaxCode1            = @PoTaxCode1
         , @HdrTaxCode2            = @PoTaxCode2
   
         , @Amount                 = @TtRcvTcCprUnitMatCost
         , @UndiscAmount           = @TtRcvTcCprUnitMatCost
   
         , @CurrCode               = @CurrCode
         , @ExchRate               = @PoExchRate
         , @UseExchRate            = 0
         , @Places                 = @CurrencyPlaces
         , @InvDate                = @Today
         , @TermsCode              = @PoTermsCode
   
         , @AmountWithoutTax       = @TtRcvTcCprUnitMatCost    OUTPUT
         , @UndiscAmountWithoutTax = @TtRcvTcCprUnitMatCost    OUTPUT
         , @Tax1OnAmount           = @TaxAmount1 OUTPUT
         , @Tax2OnAmount           = @TaxAmount2 OUTPUT
         , @Tax1OnUndiscAmount     = @xAmount    OUTPUT
         , @Tax2OnUndiscAmount     = @xAmount    OUTPUT
         , @Infobar                = @Infobar    OUTPUT
         , @Site = @ParmsSite
         , @DomCurrCode = @DomCurrCode
         , @IsTaxInterfaceAvailable = @IsTaxInterfaceAvailable
         , @vrtx_parm_Exists  = @vrtx_parm_Exists
         , @IsJapanInterfaceAvailable = @IsJapanInterfaceAvailable

      IF @Severity <> 0
         goto LineError
   
      SET @TaxAmount = isnull(@TaxAmount1,0) + isnull(@TaxAmount2,0)

      SET @TaxAmount1=NULL
      SET @TaxAmount2=NULL
      EXEC @Severity = dbo.TaxPriceSeparationSp
           @InvType                = 'PO'
         , @Type                   = 'I'
         , @TaxCode1               = @PoitemTaxCode1
         , @TaxCode2               = @PoitemTaxCode2
         , @HdrTaxCode1            = @PoTaxCode1
         , @HdrTaxCode2            = @PoTaxCode2
   
         , @Amount                 = @TtRcvTcCprUnitMatCostConv
         , @UndiscAmount           = @TtRcvTcCprUnitMatCostConv
   
         , @CurrCode               = @CurrCode
         , @ExchRate               = @PoExchRate
         , @UseExchRate            = 0
         , @Places                 = @CurrencyPlaces
         , @InvDate                = @Today
         , @TermsCode              = @PoTermsCode
   
         , @AmountWithoutTax       = @TtRcvTcCprUnitMatCostConv    OUTPUT
         , @UndiscAmountWithoutTax = @TtRcvTcCprUnitMatCostConv    OUTPUT
         , @Tax1OnAmount           = @TaxAmount1 OUTPUT
         , @Tax2OnAmount           = @TaxAmount2 OUTPUT
         , @Tax1OnUndiscAmount     = @xAmount    OUTPUT
         , @Tax2OnUndiscAmount     = @xAmount    OUTPUT
         , @Infobar                = @Infobar    OUTPUT
         , @Site = @ParmsSite
         , @DomCurrCode = @DomCurrCode
         , @IsTaxInterfaceAvailable = @IsTaxInterfaceAvailable
         , @vrtx_parm_Exists  = @vrtx_parm_Exists
         , @IsJapanInterfaceAvailable = @IsJapanInterfaceAvailable

      IF @Severity <> 0
         goto LineError
   
      SET @TaxAmountConv = isnull(@TaxAmount1,0) + isnull(@TaxAmount2,0)
   END

   SET @Cost = @TtRcvTcCprUnitMatCost + @TtRcvTcCprUnitDutyCost + @TtRcvTcCprUnitFreightCost + @TtRcvTcCprUnitBrokCost + @TtRcvTcCprUnitInsuranceCost + @TtRcvTcCprUnitLocFrtCost
   SET @TranCost = @TtRcvTcCprUnitMatCostConv + @TtRcvTcCprUnitDutyCostConv + @TtRcvTcCprUnitFreightCostConv + @TtRcvTcCprUnitBrokCostConv + @TtRcvTcCprUnitInsuranceCostConv + @TtRcvTcCprUnitLocFrtCostConv
   SET @HaveItem = case when (@ItemRowpointer is not null) then 1 else 0 end


   IF  @GrnHdrRowpointer is not null
      SET @CurrWhse = @GrnHdrWhse
   ELSE
      SET @CurrWhse = @TtRcvWhse

   -- try to avoid a deadlock
   select
     @ItemwhseRowpointer = itemwhse.rowpointer
   from itemwhse with (rowlock,xlock,holdlock)
   where itemwhse.whse = @CurrWhse
   and itemwhse.item = @PoitemItem

   /* Defind variable to skip SyncCoFromPoSp when do PO Receiving */
   IF @SourceCoNum IS NOT NULL AND @SourceSite IS NOT NULL
   BEGIN
      EXEC dbo.DefineVariableSp
        @VariableName  = 'SkipSyncCoFromPoSp'
      , @VariableValue = 1
      , @Infobar       = @Infobar OUTPUT
   END

   EXEC @Severity = dbo.PoReceivingTrxPostSp
        @TransDate         = @TtRcvTransdate
      , @Whse              = @CurrWhse
      , @PoNum             = @TtRcvPoNum
      , @PoLine            = @TtRcvPoLine
      , @PoRelease         = @TtRcvPoRelease
      , @QtyToReceive      = @TtRcvTcQtuToReceive   /* t-qty in old code */
      , @QtyToReject       = @TtRcvTcQtuToReject    /* t-qty-rej in old code */
      , @Tran1UM           = @TtRcvUM
      , @Return            = @TtRcvDrReturn
      , @EcCode            = @TtRcvEcCode
      , @HaveItem          = @HaveItem
      , @Item              = @PoitemItem
      , @Loc               = @TtRcvLoc
      , @Lot               = @TtRcvLot
      , @ReasonCode        = @TtRcvReasonCode
      , @Cost              = @Cost
      , @Material          = @TtRcvTcCprUnitMatCost
      , @Duty              = @TtRcvTcCprUnitDutyCost
      , @Freight           = @TtRcvTcCprUnitFreightCost
      , @Brokerage         = @TtRcvTcCprUnitBrokCost
      , @Insurance         = @TtRcvTcCprUnitInsuranceCost
      , @LocFrt            = @TtRcvTcCprUnitLocFrtCost
      , @TranCost          = @TranCost
      , @TranMaterial      = @TtRcvTcCprUnitMatCostConv
      , @TranDuty          = @TtRcvTcCprUnitDutyCostConv
      , @TranFreight       = @TtRcvTcCprUnitFreightCostConv
      , @TranBrokerage     = @TtRcvTcCprUnitBrokCostConv
      , @TranInsurance     = @TtRcvTcCprUnitInsuranceCostConv
      , @TranLocFrt        = @TtRcvTcCprUnitLocFrtCostConv
      , @FreightExchRate   = @FreightExchRate
      , @DutyExchRate      = @DutyExchRate
      , @BrokerageExchRate = @BrokerageExchRate
      , @InsuranceExchRate = @InsuranceExchRate
      , @LocFrtExchRate    = @LocFrtExchRate
      , @Consign           = @TtRcvByCons
      , @PackNum           = @TtRcvPacknum
      , @WorkKey           = @TtRcvWorkkey
      , @GRNExists         = @TtRcvGrnexists   OUTPUT
      , @GrnNum            = @TtRcvGrnNum
      , @GrnLine           = @GrnLineLoop     OUTPUT
      , @Infobar           = @Infobar          OUTPUT
      , @ImportDocId       = @TtRcvImportDocId
      , @TaxAmount         = @TaxAmount
      , @TaxAmountConv     = @TaxAmountconv
      , @ManufacturerId    = @TtrcvManufacturerId
      , @ManufacturerItem  = @TtrcvManufacturerItem
      , @ContainerNum      = @TrRcvContainerNum
      , @VendInvNum        = @VendInvNum
      , @DocumentNum       = @DocumentNum

   IF @SourceCoNum IS NOT NULL AND @SourceSite IS NOT NULL
   BEGIN
      EXEC dbo.UndefineVariableSp
        @VariableName  = 'SkipSyncCoFromPoSp'
      , @Infobar       = @Infobar OUTPUT
   END

   IF @Severity = 0
   BEGIN
      IF @TtRcvGrnexists <> 0
      BEGIN
         SET @GRNNowExists = 1

         IF @GrnLineLoop <> @TtRcvGrnLine
			 UPDATE tt_rcv
			 SET tt_rcv.grn_line = @GrnLineLoop
			 WHERE tt_rcv.RowPointer = @TtRcvRowPointer
			 AND tt_rcv.SessionID = @SessionID
			 AND tt_rcv.grn_num = @TtRcvGrnNum
			 AND tt_rcv.po_num = @TtRcvPoNum
			 AND tt_rcv.po_line = @TtRcvPoLine
			 AND tt_rcv.po_release = @TtRcvPoRelease
			 AND tt_rcv.sequence = @TtRcvSequence
      END

      -- qcs >>>
        IF @IsQCLicensed = 1 
        BEGIN
           if @RSQCNeedsQC = 1
           AND @TtRcvDrReturn = 0
           AND @TtRcvTcQtuToReceive > 0.0
           BEGIN
            SET @RSQCMessages = ''
            IF OBJECT_ID(N'dbo.RSQC_CreateSupplierReceiverSp') IS NOT NULL
            BEGIN
               SET @QCS_SpName = N'dbo.RSQC_CreateSupplierReceiverSp'
               EXEC @QCS_Severity = @QCS_SpName
                @TtRcvTcQtuToReceive
               , @TtRcvWhse
               , @TtRcvPoNum
               , @TtRcvPoLine
               , @TtRcvPoRelease
               , @TtRcvLoc
               , @RSQCOrigDueDate
               , @TtRcvLot
               , @PoitemItem
               , @PoVendNum
               , @TtRcvTransdate
               , @RSQCAutoAccept
               , @RSQCCallingFunction
               , '' -- @QCLot
               , @RSQCUserCode
               , 0
               , @RSQCRcvrNum OUTPUT
               , @RSQCPopUp OUTPUT
               , @RSQCPrintTag OUTPUT
               , @RSQCMessages OUTPUT
               , @Infobar OUTPUT
            END
            IF @RSQCRcvrNum IS NOT NULL
            BEGIN            
            EXEC @Severity = dbo.GetVariableSp
                 'RSQCRcvrNumStack'
                 , NULL
                 , 0
                 , @RSQCRcvrNumStack OUTPUT
                 , @Infobar OUTPUT
               EXEC @Severity = dbo.GetVariableSp
                 'RSQCStatusStack'
                 , NULL
                 , 0
                 , @RSQCStatusStack OUTPUT
                 , @Infobar OUTPUT
               EXEC @Severity = dbo.GetVariableSp
                 'RSQCMessageStack'
                 , NULL
                 , 0
                 , @RSQCMessageStack OUTPUT
                 , @Infobar OUTPUT
               EXEC @Severity = dbo.GetVariableSp
                 'RSQCSequenceStack'
                 , NULL
                 , 0
                 , @RSQCSequenceStack
                 , @Infobar OUTPUT
            
            
               IF @RSQCRcvrNumStack IS NULL 
                  SET @RSQCRcvrNumStack = CAST(@RSQCRcvrNum AS nvarchar) + ','
               ELSE
                  SET @RSQCRcvrNumStack = @RSQCRcvrNumStack + CAST(@RSQCRcvrNum AS nvarchar) + ','
                 
               IF @RSQCStatusStack IS NULL
                  IF @RSQCAutoAccept = 0
                     SET @RSQCStatusStack = 'Received,'
                  ELSE
                     SET @RSQCStatusStack = 'Accepted,'
               ELSE
                  IF @RSQCAutoAccept = 0
                     SET @RSQCStatusStack = @RSQCStatusStack + 'Received,'
                 ELSE
                   SET @RSQCStatusStack = @RSQCStatusStack + 'Accepted,'
               IF @RSQCMessageStack IS NULL
                  SET @RSQCMessageStack = @RSQCMessages
               ELSE
                  EXEC dbo.MsgConcatSp @RSQCMessageStack OUTPUT, @RSQCMessages
            
               IF @RSQCSequenceStack IS NULL
                  SET @RSQCSequenceStack = CAST(@TtRcvSequence AS nvarchar) + ','
               ELSE
                  SET @RSQCSequenceStack = @RSQCSequenceStack + CAST(@TtRcvSequence AS nvarchar) + ','
             END -- IF @RSQCRcvrNum is not null 
           END -- IF @RSQCNeedsQC, and not DR, and positive receipt 
        END-- <<< qcs
      GOTO NextLine
   end

   break

   NextLine:
   goto Commit_Data
   NextLineDone:
end
close tt_rcv_crs
deallocate tt_rcv_crs

LineError:
if @Severity > 0
begin
   if @NeedGrnInError = 1
   and @TtRcvGrnNum is null
   begin
      select top 1 @TtRcvGrnNum = po_rcpt.grn_num
      , @GrnLineLoop = po_rcpt.grn_line
      from po_rcpt with (readuncommitted)
      where po_rcpt.po_num = @TtRcvPoNum
      and po_rcpt.po_line = @TtRcvPoLine
      and po_rcpt.po_release = @TtRcvPoRelease
      and po_rcpt.grn_num is not null

      if @TtRcvGrnNum is not null
         EXEC @Severity = dbo.MsgAppSp @Infobar OUTPUT, 'E=Exist5'
         , '@grn_line'
         , '@grn_line.grn_num'
         , @TtRcvGrnNum
         , '@grn_line.grn_line'
         , @GrnLineLoop
         , '@grn_line.po_num'
         , @TtRcvPoNum
         , '@grn_line.po_line'
         , @TtRcvPoLine
         , '@grn_line.po_release'
         , @TtRcvPoRelease
   end

    if @FirstSequenceWithError is null
       SET @FirstSequenceWithError = @TtRcvSequence

    EXEC @Severity2 = dbo.JournalImmediateSp
      @Partition = @BufferJournal
    , @Infobar = @Infobar output

   if @InitialTranCount = 0
      ROLLBACK TRANSACTION

   UPDATE tt_rcv
   SET PostSuccess = 0
   WHERE RowPointer = @TtRcvRowPointer
   and tt_rcv.SessionID = @SessionID  
   and tt_rcv.grn_line = @GrnLineLoop
   and tt_rcv.po_num = @TtRcvPoNum
   and tt_rcv.po_line = @TtRcvPoLine
   and tt_rcv.po_release = @TtRcvPoRelease
   and tt_rcv.sequence = @TtRcvSequence
end
else if @@trancount > @InitialTranCount
begin
Commit_Data:
   EXEC @Severity2 = dbo.JournalImmediateSp
     @Partition = @BufferJournal
    ,@Infobar = @Infobar output
   if @Severity = 0
      set @Severity = @Severity2

   if @Severity = 0
   begin
      update tt_rcv
      set PostSuccess = 1
      WHERE RowPointer = @TtRcvRowPointer
      and tt_rcv.SessionID = @SessionID
      and tt_rcv.grn_line = @GrnLineLoop
      and tt_rcv.po_num = @TtRcvPoNum
      and tt_rcv.po_line = @TtRcvPoLine
      and tt_rcv.po_release = @TtRcvPoRelease
      and tt_rcv.sequence = @TtRcvSequence

      IF @PoitemFaNum IS NOT NULL
         update famaster 
         set acq_date = @TtRcvTransdate
         where fa_num = @PoitemFaNum and fa_stat =  N'N'

      if @InitialTranCount = 0
         COMMIT TRANSACTION

      SET @Infobar = NULL
      SET @Severity = 0

      -- if the cursor is still open, then there must be more work to do

      and tt_rcv.SessionID = @SessionID  
      and tt_rcv.grn_line = @GrnLineLoop
      and tt_rcv.po_num = @TtRcvPoNum

      and tt_rcv.po_release = @TtRcvPoRelease
